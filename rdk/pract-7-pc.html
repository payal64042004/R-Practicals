  PC         = paste0("PC", 1:length(v_pc)),
  Eigenvalue = v_pc,
  Proportion = pv_pc,
  Cumulative = cpv_pc
)

# PCA on Iris data
#==========================================
library(ggplot2)
data(iris)

Y = (iris[, 1:4])
pca_iris = prcomp(Y, scale. = TRUE)
summary(pca_iris)
## Importance of components:
##                           PC1    PC2     PC3     PC4
## Standard deviation     1.7084 0.9560 0.38309 0.14393
## Proportion of Variance 0.7296 0.2285 0.03669 0.00518
## Cumulative Proportion  0.7296 0.9581 0.99482 1.00000
#-------------------------------------------
# Eigenvalues 
#--------------------------------------
eigenvalues = pca_iris$sdev^2 # Variance of PC's
pv_pc = eigenvalues / sum(eigenvalues) * 100
print(round(pv_pc, 2))
## [1] 72.96 22.85  3.67  0.52
cpv_pc = cumsum(pv_pc)

# Scree Plot
plot(pv_pc, type = "b", pch = 19, col = "blue",
     xlab = "Principal Component", ylab = "Variance Explained (%)",
     main = "Scree Plot of Iris PCA")
